// AdminDashboard.js
import React, { useEffect, useState } from "react";
import { Container, Row, Col, Card, Button, Table, Badge } from "react-bootstrap";
import { useNavigate } from "react-router-dom";
import Swal from "sweetalert2";
import axios from "../api/axios";
import { FaBox, FaShoppingCart, FaUsers, FaDollarSign } from "react-icons/fa";

function AdminDashboard() {
  const navigate = useNavigate();
  const name = localStorage.getItem("name") || "Admin";

  const [stats, setStats] = useState({
    products: 0,
    orders: 0,
    users: 0,
    revenue: 0,
  });
  const [latestOrders, setLatestOrders] = useState([]);

  // Fetch stats and latest orders
  useEffect(() => {
    async function fetchData() {
      try {
        const productRes = await axios.get("/products");
        const orderRes = await axios.get("/orders");
        const userRes = await axios.get("/users");

        setStats({
          products: productRes.data.length || 0,
          orders: orderRes.data.length || 0,
          users: userRes.data.length || 0,
          revenue: orderRes.data.reduce((sum, o) => sum + (o.totalAmount || 0), 0),
        });

        setLatestOrders(orderRes.data.slice(0, 5));
      } catch (error) {
        console.error(error);
      }
    }
    fetchData();
  }, []);

  const handleLogout = () => {
    Swal.fire({
      title: "Are you sure?",
      text: "You will be logged out!",
      icon: "warning",
      showCancelButton: true,
      confirmButtonColor: "#3085d6",
      cancelButtonColor: "#d33",
      confirmButtonText: "Yes, logout",
    }).then((result) => {
      if (result.isConfirmed) {
        localStorage.clear();
        Swal.fire("Logged Out!", "You have been logged out successfully.", "success");
        navigate("/login");
      }
    });
  };

  return (
    <Container fluid style={{ marginTop: "80px" }}>
      <h2 className="mb-4 text-center">Welcome, {name} 👋</h2>

      {/* Stats Row */}
      <Row className="g-4 mb-4">
        <Col md={3}>
          <Card className="shadow-sm text-center p-3 bg-light">
            <FaBox size={30} className="mb-2 text-primary" />
            <h5>Total Products</h5>
            <h3>{stats.products}</h3>
          </Card>
        </Col>
        <Col md={3}>
          <Card className="shadow-sm text-center p-3 bg-light">
            <FaShoppingCart size={30} className="mb-2 text-success" />
            <h5>Total Orders</h5>
            <h3>{stats.orders}</h3>
          </Card>
        </Col>
        <Col md={3}>
          <Card className="shadow-sm text-center p-3 bg-light">
            <FaUsers size={30} className="mb-2 text-warning" />
            <h5>Total Users</h5>
            <h3>{stats.users}</h3>
          </Card>
        </Col>
        <Col md={3}>
          <Card className="shadow-sm text-center p-3 bg-light">
            <FaDollarSign size={30} className="mb-2 text-danger" />
            <h5>Total Revenue</h5>
            <h3>₹{stats.revenue}</h3>
          </Card>
        </Col>
      </Row>

      {/* Latest Orders Table */}
      <Row>
        <Col md={8}>
          <Card className="shadow-sm">
            <Card.Header><strong>Latest Orders</strong></Card.Header>
            <Card.Body>
              <Table hover responsive>
                <thead>
                  <tr>
                    <th>#</th>
                    <th>Customer</th>
                    <th>Total</th>
                    <th>Status</th>
                  </tr>
                </thead>
                <tbody>
                  {latestOrders.length > 0 ? (
                    latestOrders.map((order, index) => (
                      <tr key={order._id}>
                        <td>{index + 1}</td>
                        <td>{order.customerName || "N/A"}</td>
                        <td>₹{order.totalAmount}</td>
                        <td>
                          <Badge
                            bg={
                              order.status === "Delivered"
                                ? "success"
                                : order.status === "Pending"
                                ? "warning"
                                : "secondary"
                            }
                          >
                            {order.status}
                          </Badge>
                        </td>
                      </tr>
                    ))
                  ) : (
                    <tr>
                      <td colSpan="4" className="text-center text-muted">
                        No recent orders
                      </td>
                    </tr>
                  )}
                </tbody>
              </Table>
            </Card.Body>
          </Card>
        </Col>

        {/* Quick Actions */}
        <Col md={4}>
          <Card className="shadow-sm mb-3 text-center p-3">
            <h5>Quick Actions</h5>
            <Button className="m-2" variant="primary" onClick={() => navigate("/manageproducts")}>
              Manage Products
            </Button>
            <Button className="m-2" variant="success" onClick={() => navigate("/manageorders")}>
              Manage Orders
            </Button>
            <Button className="m-2" variant="danger" onClick={handleLogout}>
              Logout
            </Button>
          </Card>
        </Col>
      </Row>
    </Container>
  );
}

export default AdminDashboard;
